CMAKE_MINIMUM_REQUIRED (VERSION 2.6)
PROJECT (SimpleGL)

# standart
INCLUDE(Configure.cmake)

# vars 
SET (TARGET_NAME 		SimpleGL)
SET (TARGET_UNIX_NAME 	sgl)

# sources
ADD_SUBDIRECTORY (src)

# examples 
IF (BUILD_EXAMPLES)
	ADD_SUBDIRECTORY (examples)
ENDIF (BUILD_EXAMPLES)

IF (SIMPLE_GL_BUILD_DOCUMENTATION)
    INCLUDE(Doxy)
ENDIF (SIMPLE_GL_BUILD_DOCUMENTATION)

# uninstall rule
IF (NOT SIMPLE_GL_CONFIGURE_INTRUSIVE)
	INCLUDE(Uninstall OPTIONAL)
ENDIF (NOT SIMPLE_GL_CONFIGURE_INTRUSIVE)
